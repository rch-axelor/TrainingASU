import java.io.*;

class Encapsulate {
    private String VechicleName;
    private String VechicleNumber;
    private int VechicleAge;
    public int getVechicleAge() { return VechicleAge; }
    public String getVechicleName() { return VechicleName; }
    public String getVechicleNumber() { return VechicleNumber; }
    public void setVechicleAge(int newAge) { VechicleAge = newAge; }
    public void setVechicleName(String newName){ VechicleName = newName; }
    public void setVechicleNumber(String newNumber) { VechicleNumber = newNumber; }
}
abstract class Vechicle{
  public abstract void VechicleType();
  public void colour() {
    System.out.println("Pink");
  }
}
class Luxury extends Vechicle{
  public void VechicleType() {
    System.out.println("Luxury");
  }
}
public class VechicleSum {
    public int sum(int x, int y){ return (x + y); }
    public int sum(int x, int y, int z){ return (x + y + z); }
    public double sum(double x, double y){ return (x + y); }
}

    BMW.VechicleType();

  }
